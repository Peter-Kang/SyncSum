def address
  if street.blank? and city.blank? and state.blank?
    "(Empty Address)"
  elsif street.blank? and city.blank?
    "#{state}"
  elsif street.blank? and state.blank?
    "#{city}"
  elsif city.blank? and state.blank?
    "#{street}"
  elsif street.blank?
    "#{city}, #{state}"
  elsif city.blank?
    "#{street}, #{state}"
  elsif state.blank?
    "#{street}, #{city}"
  end
end
