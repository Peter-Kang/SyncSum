def add_answer_for_question(answer)
  answered_questions.each do |question|
    if question.in_answers?(answer)
      temp = nil
      self.answers.each do |a|
        temp = a if question.in_answers?(a)
      end
      self.answers.delete(temp)
    end
  end
  self.answers << answer
end
